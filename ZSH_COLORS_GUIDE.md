# ============================================
# НАСТРОЙКА ЦВЕТОВ В ZSH
# ============================================

## 1. Открой конфигурацию zsh
nano ~/.zshrc

## 2. Добавь в конец файла:

# ============================================
# ЦВЕТА ПРОМПТА (строка ввода команд)
# ============================================

# Включить цвета
autoload -U colors && colors

# Простой цветной промпт (зеленый)
PROMPT='%F{green}%n@%m%f:%F{blue}%~%f$ '

# Или красный
PROMPT='%F{red}%n@%m%f:%F{yellow}%~%f$ '

# Или многоцветный (как у профи)
PROMPT='%F{cyan}┌─[%f%F{green}%n%f%F{cyan}@%f%F{magenta}%m%f%F{cyan}]─[%f%F{yellow}%~%f%F{cyan}]
└─%f$ '

# ============================================
# ЦВЕТА ВЫВОДА КОМАНД
# ============================================

# Цвета для ls (список файлов)
export CLICOLOR=1
export LSCOLORS=ExGxBxDxCxEgEdxbxgxcxd

# Для Linux (если используешь WSL)
alias ls='ls --color=auto'
alias grep='grep --color=auto'

# ============================================
# ЦВЕТА ДЛЯ ВЫВОДА ПРОГРАММ
# ============================================

# Зеленый текст по умолчанию (для всего терминала)
echo -ne "\033[32m"

# Красный текст по умолчанию
echo -ne "\033[31m"

# Сбросить цвет (вернуть белый)
echo -ne "\033[0m"

## 3. Применить изменения
source ~/.zshrc

# ============================================
# КОДЫ ЦВЕТОВ
# ============================================

# В PROMPT (используй %F{цвет}):
# - black, red, green, yellow, blue, magenta, cyan, white

# В echo (используй \033[XXm):
# 30 - черный
# 31 - красный
# 32 - зеленый
# 33 - желтый
# 34 - синий
# 35 - пурпурный
# 36 - голубой
# 37 - белый
# 0  - сброс (по умолчанию)

# Яркие цвета (90-97):
# 91 - ярко-красный
# 92 - ярко-зеленый
# 93 - ярко-желтый
# 94 - ярко-синий

# ============================================
# ПРИМЕРЫ КРУТЫХ ПРОМПТОВ
# ============================================

# Минималистичный зеленый
PROMPT='%F{green}➜%f '

# Как у профи (с git веткой)
autoload -Uz vcs_info
precmd() { vcs_info }
zstyle ':vcs_info:git:*' formats '%F{yellow}(%b)%f'
setopt PROMPT_SUBST
PROMPT='%F{green}%n%f %F{blue}%~%f ${vcs_info_msg_0_} %F{red}❯%f '

# Двухстрочный (как у хакеров)
PROMPT='%F{cyan}╭─%f%F{green}%n@%m%f %F{yellow}%~%f
%F{cyan}╰─%f%F{red}$%f '

# ============================================
# УСТАНОВКА OH-MY-ZSH (рекомендую!)
# ============================================

# Oh-My-Zsh - фреймворк для zsh с готовыми темами
sh -c "$(curl -fsSL https://raw.githubusercontent.com/ohmyzsh/ohmyzsh/master/tools/install.sh)"

# После установки выбери тему в ~/.zshrc:
ZSH_THEME="agnoster"  # Популярная тема
# или
ZSH_THEME="robbyrussell"  # Простая и быстрая
# или
ZSH_THEME="powerlevel10k/powerlevel10k"  # Самая крутая

# Применить
source ~/.zshrc

# ============================================
# ЦВЕТА ДЛЯ ВЫВОДА C++ ПРОГРАММ
# ============================================

# В твоем C++ коде:
#include <iostream>
using namespace std;

int main() {
    // Красный текст
    cout << "\033[31mКрасный текст\033[0m" << endl;
    
    // Зеленый текст
    cout << "\033[32mЗеленый текст\033[0m" << endl;
    
    // Синий текст
    cout << "\033[34mСиний текст\033[0m" << endl;
    
    // Жирный красный
    cout << "\033[1;31mЖирный красный\033[0m" << endl;
    
    return 0;
}
